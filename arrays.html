Apply

		var array = [
			{name: "Google", url: "http://www.google.com"},
			{name: "Yahoo", url: "http://www.yahoo.com"}
		];
		<ul id="list"/>
		Arrays.apply(array, function(xi, i, n) {
			$("#list").append('<li><a href="'+xi.url+'">'+xi.name+'</a></li>');
		});

Range

		var x = Arrays.range(3, 10);

Filling

		var y = Arrays.fill(7, function(i, n) {
			return i*i;
		});

Transform

		var z = Arrays.transform(y, function(input) {
			return input / 2.0;
		});

Filtering

		var q = Arrays.filter(z, function(input) {
			return input >= 5;
		});

Aggregation

		var ag = Arrays.aggregate(q, 0, function(a, xi, i) {
			return a + xi;
		});

Search

		var f = Arrays.find(q, function(xi, i) {
			return xi >= 10;
		});
